CREATE TABLE customer (
    id INT AUTO_INCREMENT PRIMARY KEY,
    business_name VARCHAR(255) NOT NULL,
    owner_name VARCHAR(255) NOT NULL,
    meeting_purpose VARCHAR(255) NOT NULL,
    business_type VARCHAR(255) NOT NULL,
    unique_selling_product VARCHAR(255) NOT NULL,
    phone_contact_1 VARCHAR(20) NOT NULL,
    phone_contact_2 VARCHAR(20),
    business_location VARCHAR(255) NOT NULL,
    address TEXT NOT NULL,
    website_url VARCHAR(255),
    business_card VARCHAR(255),
    email VARCHAR(255) NOT NULL,
    customer_type VARCHAR(50) NOT NULL,
    industry VARCHAR(255),
    company_size VARCHAR(50),
    annual_revenue DECIMAL(15, 2),
    lead_source VARCHAR(255),
    contact_preference VARCHAR(50),
    notes TEXT,
    status VARCHAR(50) NOT NULL,
    assigned_sales_rep VARCHAR(255),
    last_contacted DATETIME,
    date_created DATETIME DEFAULT CURRENT_TIMESTAMP,
    date_edited DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
